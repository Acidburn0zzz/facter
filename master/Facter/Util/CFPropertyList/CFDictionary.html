def to_xml(parser)
  n = parser.new_node('dict')
  @value.each_pair do |key, value|
    k = parser.append_node(parser.new_node('key'), parser.new_text(key.to_s))
    n = parser.append_node(n, k)
    n = parser.append_node(n, value.to_xml(parser))
  end
  n
end
